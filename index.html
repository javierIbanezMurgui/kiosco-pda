// --- CONFIGURACIÓN DE APIS --- const URL_USUARIOS = "https://script.google.com/macros/s/AKfycbyuYH3otEfAyqFEfr9n5ed4yxQ8eZK3Eal8QVWk7l6XPPmr_pXkjPU2toCNJR6WSJw/exec?fn=getUsuarios"; const URL_DISPOSITIVOS = "https://script.google.com/macros/s/AKfycbyuYH3otEfAyqFEfr9n5ed4yxQ8eZK3Eal8QVWk7l6XPPmr_pXkjPU2toCNJR6WSJw/exec?fn=getDispositivos"; const URL_LOGIN = "https://script.google.com/macros/s/AKfycbyuYH3otEfAyqFEfr9n5ed4yxQ8eZK3Eal8QVWk7l6XPPmr_pXkjPU2toCNJR6WSJw/exec?sheet=login"; const URL_LOGOUT = "https://script.google.com/macros/s/AKfycbyuYH3otEfAyqFEfr9n5ed4yxQ8eZK3Eal8QVWk7l6XPPmr_pXkjPU2toCNJR6WSJw/exec?sheet=logout"; let NUMERO_PDA = null; let DEVICE_ID_REAL = "NO_DETECTADO"; let USUARIO_ACTUAL = null; function getDeviceId() { if (typeof freekiosk !== "undefined" && freekiosk.getDeviceId) return freekiosk.getDeviceId(); return "PDA_TEST"; } async function inicializar() { DEVICE_ID_REAL = getDeviceId(); document.getElementById("pdaLabel").innerText = "Conectando..."; document.getElementById("loader").style.display = "block"; try { const res = await fetch(`${URL_DISPOSITIVOS}&cb=${Date.now()}`); if (!res.ok) throw new Error("Error de servidor"); const data = await res.json(); const pdaFound = data.dispositivos.find(d => String(d.device).trim() === String(DEVICE_ID_REAL).trim()); if (pdaFound) { NUMERO_PDA = pdaFound.numeroPda; document.getElementById("pdaLabel").innerText = "PDA Nº " + NUMERO_PDA; document.getElementById("loginButton").disabled = false; } else { document.getElementById("pdaLabel").innerText = "SIN REGISTRO"; mostrarError("Este dispositivo no está registrado: " + DEVICE_ID_REAL); } } catch (err) { document.getElementById("pdaLabel").innerText = "ERROR DE RED"; mostrarError("Verifica la conexión Wi-Fi: " + err.message); } finally { document.getElementById("loader").style.display = "none"; } } async function login() { mostrarError(""); const u = document.getElementById("usuario").value.trim(); const p = document.getElementById("pin").value.trim(); if (!u || !p) return mostrarError("Introduce usuario y PIN"); if (!NUMERO_PDA) return mostrarError("Error: PDA no identificada"); document.getElementById("loader").style.display = "block"; try { const res = await fetch(URL_USUARIOS); const data = await res.json(); const user = data.usuarios.find(x => x.usuario === u && String(x.pin) === p); if (user) { if (String(user.pdaAsignada).trim() !== String(NUMERO_PDA).trim()) { mostrarError(`No es tu PDA: Tu PDA asignada es la Nº ${user.pdaAsignada}`); document.getElementById("loader").style.display = "none"; return; } USUARIO_ACTUAL = user.usuario; document.getElementById("userTag").innerText = "Operario: " + USUARIO_ACTUAL; document.getElementById("loginDiv").style.display = "none"; document.getElementById("menuDiv").style.display = "block"; // Registrar login fetch(URL_LOGIN, { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify({ fecha: new Date().toLocaleString(), usuario: USUARIO_ACTUAL, numeroPda: NUMERO_PDA, device: DEVICE_ID_REAL, lat: "", lng: "" }) }); } else { mostrarError("Usuario o PIN incorrecto"); } } catch (e) { mostrarError("Error al validar datos. Reintente."); } finally { document.getElementById("loader").style.display = "none"; } } async function cerrarTurno() { if(!confirm("¿Deseas finalizar el turno y cerrar sesión?")) return; const label = document.getElementById("pdaLabel"); label.innerText = "CERRANDO SESIÓN..."; document.getElementById("menuDiv").style.display = "none"; document.getElementById("loader").style.display = "block"; try { await fetch(URL_LOGOUT, { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify({ fecha: new Date().toLocaleString(), usuario: USUARIO_ACTUAL, numeroPda: NUMERO_PDA, device: DEVICE_ID_REAL, lat: "", lng: "" }) }); } catch(e) { console.error("Error al registrar logout:", e); } finally { location.reload(); } } function abrirApp(pkg) { if (typeof freekiosk !== "undefined" && freekiosk.launchApp) { freekiosk.launchApp(pkg); } else { alert("Simulando apertura de: " + pkg); } } window.onload = () => { setTimeout(inicializar, 1000); };
