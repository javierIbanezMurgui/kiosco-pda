function getDeviceId() { if (typeof FreeKiosk !== "undefined" && FreeKiosk.getDeviceId) return FreeKiosk.getDeviceId(); return "TEST_DEVICE_01"; // Para pruebas en PC } function showError(msg) { const e = document.getElementById("error"); e.innerText = msg || ""; e.style.display = msg ? "block" : "none"; } async function inicializar() { DEVICE_ID = getDeviceId(); try { const resp = await fetch(`${API}?action=init&device=${DEVICE_ID}`); const data = await resp.json(); if (data.status === "ok") { NUMERO_PDA = data.numeroPda; document.getElementById("pdaLabel").innerText = "PDA Nº " + NUMERO_PDA; document.getElementById("loginBtn").disabled = false; } else { document.getElementById("pdaLabel").innerText = "Dispositivo NO registrado"; showError("Este terminal no tiene permiso de acceso."); } } catch (e) { document.getElementById("pdaLabel").innerText = "Error de conexión"; showError("No se pudo conectar con el servidor."); } } async function getCoords() { return new Promise((resolve) => { navigator.geolocation.getCurrentPosition( (p) => resolve({ lat: p.coords.latitude, lng: p.coords.longitude }), () => resolve({ lat: 0, lng: 0 }), { timeout: 5000 } ); }); } async function login() { showError(""); const user = document.getElementById("usuario").value; const pin = document.getElementById("pin").value; const btn = document.getElementById("loginBtn"); if (!user || !pin) return showError("Completa todos los campos"); btn.disabled = true; btn.innerText = "Verificando..."; const coords = await getCoords(); try { const response = await fetch(API, { method: "POST", body: JSON.stringify({ login: { usuario: user, pin: pin, device: DEVICE_ID, pda: NUMERO_PDA, lat: coords.lat, lng: coords.lng } }) }); const res = await response.json(); if (res.status === "ok") { USUARIO_ACTUAL = user; document.getElementById("login").style.display = "none"; document.getElementById("menu").style.display = "block"; document.getElementById("usuarioActivo").innerText = "Operario: " + user; } else { showError(res.msg); } } catch (e) { showError("Error de red al intentar loguear."); } finally { btn.disabled = false; btn.innerText = "Iniciar sesión"; } } function abrirUrl() { window.location.href = "https://www.marca.com/"; } function abrirApp(pkg) { if (typeof FreeKiosk !== "undefined" && FreeKiosk.launchApp) FreeKiosk.launchApp(pkg); else alert("Función solo disponible en la PDA"); } async function logout() { const coords = await getCoords(); await fetch(API, { method: "POST", body: JSON.stringify({ logout: { usuario: USUARIO_ACTUAL, pda: NUMERO_PDA, device: DEVICE_ID, lat: coords.lat, lng: coords.lng } }) }); location.reload(); } window.onload = () => { setTimeout(inicializar, 500); };
